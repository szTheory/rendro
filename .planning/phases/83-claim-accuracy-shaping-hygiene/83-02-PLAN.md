From lib/rendro/text/bidi.ex — CURRENT resolve_state/1 (lines 63–84, the only section changing):
```elixir
defp resolve_state(cp) do
  script_name = UnicodeData.Script.script_from_codepoint(cp)
  # returns String: "Arabic", "Latin", "Common", "Inherited", "Unknown"

  script_tag =
    if script_name in ["Common", "Inherited", "Unknown"] do
      :common
    else
      script_name |> UnicodeData.Script.script_to_tag() |> String.to_atom()
      # script_to_tag returns lowercase 4-letter string: "arab", "latn", "hebr"
    end

  bidi_class = UnicodeData.Bidi.bidi_class(cp)
  # returns String: "L", "R", "AL", etc.

  direction =
    case bidi_class do
      "L" -> :ltr
      "R" -> :rtl
      "AL" -> :rtl
      _ -> :neutral
    end

  %{script: script_tag, direction: direction}
end
```
