```elixir
defp find_executable do
  finder = Application.get_env(:rendro, :pdfium_cli_executable_finder, &default_finder/1)
  case finder.("pdfium-cli") || finder.("pdfium") do
    nil -> {:error, {:missing_executable, "pdfium-cli"}}
    executable -> {:ok, executable}
  end
end

defp run_command(executable, args, opts \\ []) do
  runner = Application.get_env(:rendro, :pdfium_cli_command_runner, &System.cmd/3)
  cmd_opts = Keyword.get(opts, :cmd_opts, stderr_to_stdout: true)
  try do
    case runner.(executable, args, cmd_opts) do
      {output, 0} -> {:ok, output}
      {output, exit_code} -> {:error, {:pdfium_cli_failed, exit_code, output}}
    end
  rescue
    error -> {:error, {:command_failed, error.__struct__}}
  end
end
```

From lib/rendro/adapters/qpdf.ex (tmp-dir + write_private_file pattern):
```elixir
defp make_tmp_dir do
  path = Path.join(System.tmp_dir!(), "rendro-protect-#{System.unique_integer([:positive, :monotonic])}")
  with :ok <- File.mkdir_p(path), :ok <- File.chmod(path, 0o700) do
    {:ok, path}
  else
    {:error, reason} -> {:error, reason}
  end
end

defp write_private_file(path, contents) do
  File.rm(path)
  with :ok <- File.write(path, contents, [:write, :exclusive, :binary]),
       :ok <- File.chmod(path, 0o600) do
    :ok
  end
end
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| caller → render/2 | pdf_binary and opts values are untrusted; dpi/pages must be validated before passing to CLI |
| render/2 → pdfium-cli | CLI invoked via System.cmd/3 with list-form args — no shell; tmp file written 0o600 |
| tmp dir → filesystem | Isolated per-invocation dir (chmod 0o700) prevents cross-invocation file reads |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-85-04 | Info Disclosure | input.pdf written to tmp file | mitigate | write_private_file/2 uses [:write, :exclusive, :binary] + File.chmod(path, 0o600); tmp dir is chmod 0o700; File.rm_rf in after block per qpdf.ex pattern |
| T-85-05 | Tampering | pdfium-cli binary injection via :pdfium_cli_executable_finder | accept | injectable finder exists for testing only; production uses System.find_executable; no user-facing API exposes the injection |
| T-85-06 | Elevation of Privilege | command injection via dpi/pages args | mitigate | dpi validated as positive integer (Integer.to_string); pages validated as non-empty string; both passed as discrete list elements to System.cmd/3 (no shell expansion); no user-controlled string interpolated into shell command |
| T-85-SC | Tampering | npm/pip/cargo installs | accept | No new packages installed; pdfium-cli binary already addressed by T-85-07 in Plan 04 |
